function calc(){
    var inp1 = prompt("Enter 1st number")
    var inp2 = prompt("Enter 2nd number")
    var op = prompt("Enter operator")
    if (op === "+"){
        document.write(inp1 + "+" + inp2 + "=" + (+inp1+ +inp2))
    }
    else if (op === "-"){
        console.log(inp1 + "-" + inp2 + "=" + inp1 -inp2)
    }
    else if (op === "*"){
        document.write(inp1 + "*" + inp2 + "=" + inp1 * inp2)
    }
    else if (op === "/"){
        document.write(inp1 + "/" + inp2 + "=" + inp1 / inp2)
    }
    else if (op === "%"){
        document.write(inp1 + "%" + inp2 + "=" + inp1 % inp2)
    }
}
calc()
